CREATE TABLE Users (
    Id UNIQUEIDENTIFIER PRIMARY KEY,
    Name NVARCHAR(70),
	FirstName NVARCHAR(30),
	LastName NVARCHAR(30),
    Email NVARCHAR(100),
	Username NVARCHAR(100),
	Password NVARCHAR(12),
	Photo NVARCHAR(MAX),
	Bio NVARCHAR(500),
	CreatedOn DATETIME,
	DeletedOn DATETIME,
	Deleted CHAR
);


CREATE TABLE Tweet (
    Id NVARCHAR(100),
	Owner NVARCHAR(100),
	OwnerId NVARCHAR(100),
    Description NVARCHAR(280),
	Likes INT,
	Retweets INT,
	CreatedOn DATETIME,
	DeletedOn DATETIME,
	Deleted CHAR
);

CREATE TABLE ReTweet (
	Id UNIQUEIDENTIFIER,
	TweetId UNIQUEIDENTIFIER,
	ReTweetedBy UNIQUEIDENTIFIER,
	CreatedOn DATETIME,
	DeletedOn DATETIME,
	Deleted CHAR
);

CREATE TABLE Likes (
	Id UNIQUEIDENTIFIER,
	TweetId UNIQUEIDENTIFIER,
	LikedBy UNIQUEIDENTIFIER,
	CreatedOn DATETIME,
	DeletedOn DATETIME,
	Deleted CHAR
);

CREATE TABLE Comment (
	Id UNIQUEIDENTIFIER,
	TweetId UNIQUEIDENTIFIER,
	CommentBy UNIQUEIDENTIFIER,
	CreatedOn DATETIME,
	DeletedOn DATETIME,
	Deleted CHAR
);

CREATE TABLE Follow (
	Id UNIQUEIDENTIFIER,
	UserId UNIQUEIDENTIFIER,
	FollowedBy UNIQUEIDENTIFIER,
	CreatedOn DATETIME,
	DeletedOn DATETIME,
	Deleted CHAR
);
